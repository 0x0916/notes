#include <stdio.h>
#include <stdlib.h>
#include <time.h>
#include <unistd.h>
#include <string.h>

void errExit(char *err) {
	perror(err);
	exit(EXIT_FAILURE);
}

char *currTime(const char *format) {
	static char buf[1000];
	time_t	t;
	size_t	s;
	struct tm *tm;

	t = time(NULL);
	tm = localtime(&t);
	if (tm == NULL)
		return NULL;

	s = strftime(buf, 1000, (format != NULL) ? format : "%c", tm);

	return (s == 0) ? NULL : buf;
}

int main(int argc, char **argv) {
 
	int pipefd[2];

	if (argc < 2 || strcmp(argv[1], "--help") == 0) {
