C Code: 获取一个终端的名称

#include <stdio.h>
#include <unistd.h>
#include <stdlib.h>

int main(int argc, char **argv){
	char *name;

	if (isatty(0)) {
 
		name = ttyname(0);
		if (name == NULL)
			name = "undefined";
	}else {
		name = "not a tty";
	}
	printf("fd 0: %s\n", name);

	if (isatty(1)) {
		name = ttyname(1);
		if (name == NULL)
			name = "undefined";
	}else {
		name = "not a tty";
	}
	printf("fd 1: %s\n", name);

	if (isatty(2)) {
		name = ttyname(2);
		if (name == NULL)
			name = "undefined";
	}else {
		name = "not a tty";
	}
	printf("fd 2: %s\n", name);

	exit(1);
}

ttyname获取文件描述符上打开的终端设备的路径名

 
$ gcc -o ttyname ttyname.c 
$ ./ttyname </dev/console 2>/dev/null
fd 0: /dev/console
fd 1: /dev/pts/8
fd 2: not a tty
